;=> 直接创建map
{:a 1, :b 2}

;=> 通过键值对创建map
(into {} [[:a 1] [:b 2]])
(hash-map :a 1 :b 2)
(apply hash-map [:a 1 :b 2])
;=> {:a 1, :b 2}

; 压缩键值对生成map
(zipmap [:a :b] [1 2])
;=> {:b 2, :a 1}

; 创建一个有序map
(sorted-map "bac" 2 "abc" 9)
;=> {"abc" 9, "bac" 2}

(sorted-map-by #(compare (subs %1 1) (subs %2 1)) "bac" 2 "abc" 9)
;=> {"bac" 2, "abc" 9}
