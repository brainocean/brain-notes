;; Top key override
clj -X:my-fn :config 789

;; Nested key override
clj -X:my-fn '[:my :data]' 789

;; deps.edn
{:aliases
 {:my-fn
  {:exec-fn my.qualified/fn
   :exec-args {:my {:data 123}
               :config 456}}}}

;; 执行任意函数--不通过alias
clj -X my.qualified/fn :config 789
#+END_SRC
